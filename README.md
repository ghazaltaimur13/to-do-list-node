# Step - 1
npm install

# Step - 2
Create Database named as "todolist"

# Step - 3
To create migration we need to install sequelize-cli
npm install -g sequelize-cli

then run:
sequelize db:migrate

# Step - 4
To run seeders we need to run command:
node seeders

# Step - 5
To start the app:
npm start 
or
node app
